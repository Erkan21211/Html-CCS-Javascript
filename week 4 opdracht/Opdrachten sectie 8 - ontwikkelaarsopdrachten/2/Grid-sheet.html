Grid

Container properties

//Create a grid
display: grid || inline-grid
//Creates grid columns
grid-template-columns: length || percentage || fr || line-name
//Creates grid rows
grid-template-rows: length || percentage || fr ||line-name
//Creates a grid with specified areas
grid-template-areas: grid-area-name || none ||
//Creates rows, columns and areas in a single declaration
grid-template: none (initial values) || grid-template-rows /grid-template-columns
//Creates spaces between grid-items on the row axis
row-gap (grid-row-gap) : length
//Creates spaces between grid-itemes on the column axis
column-gap (grid-column-gap) : length
//Creates spaces between grid-items on the row and column axis
gap (grid-gap) : grid-row-gap grid-column-gap
//Align grid items along the block / row axis
justify-items: start || end || center || stretch
//Align grid items along the block / column axis
justify-items: start || end || center || stretch
//Align grid items along the inline and block axis (shorthand)
place-items: align-items / justify-items
//Align the whole grid (container) along the inline / row axis
justify-content: start || end || center || stretch || space-around || space-between || space-evenly
//Align the whole grid (container) along the inline / block axis
align-content: start || end || center || stretch || space-around || space-between || space-evenly
//Align the whole grid (container) along the inline and block axis (shorthand)
place-items: align-content / justify-content
//Specifies the size of any auto-generated grid row
grid-auto-row: size (length || percentage || fr)
//Specifies the size of any auto-generated grid column
grid-auto-columns: size (length || percentage || fr)
//Sets the algorithm that places the grid-items
grid-auto-flow: row || column || dense
//Shorthand for almost every property
grid: grid-template-rows grid-template-columns grid-template-areas grid-auto-row grid-auto-columns grid-auto-flow

Item Properties

//Determines the column start of an item within the grid
grid-column-start: line || span (number || name) || auto
//Determines the row start of an item within the grid
grid-row-start: line || span (number || name) || auto
//Determines the column end of an item within the grid
grid-column-end: line || span (number || name) || auto
//Determines the row end of an item within the grid
grid-row-end: line || span (number || name) || auto
//Shorthand for column start and end
grid-column: grid-column-start grid-column-end
//Shorthand for row start and end
grid-row: grid-row-start grid-row-end
//Give a grid area name to a grid item
grid-area: name || row-start / column-start / row-end / column-end
//Align a grid-item inside a cell along the row / inline axis
justify-self: start || end || center || stretch
//Align a grid-item inside a cell along the column / block axis
align-self: start || end || center || stretch
//Aligns a grid-item inside a cell along the row / inline and column / block axis
place-self: align-self justify-self
